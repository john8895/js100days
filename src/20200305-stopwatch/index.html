    let timer = 0;
    let Interval;
    let minute = document.getElementById('minute');
    let second = document.getElementById('second');
    let ms = document.getElementById('ms');

    function startAndStop() {
        //STOP
        if (typeof Interval !== "undefined" && timer !== 0) {
            clearInterval(Interval);
            watchText.innerText = "stop";
        } else {
            //START
            watchText.innerText = "start";
            Interval = setInterval(function () {
                timer += 1 / 60;
                let msVal = Math.floor((timer - Math.floor(timer)) * 100);
                ms.innerHTML = msVal < 10 ? "0" + msVal.toString() : msVal;

                let secondVal = Math.floor(timer) - Math.floor(timer / 60) * 60;
                second.innerHTML = secondVal < 10 ? "0" + secondVal.toString() : secondVal;

                let minuteVal = Math.floor(timer / 60);
                minute.innerHTML = minuteVal < 10 ? "0" + minuteVal.toString() + ":" : minuteVal + ":";

            }, 1000 / 60)
        }
    }

    function reset() {
        clearInterval(Interval);
        timer = 0;
        ms.innerHTML = "00";
        second.innerHTML = "00";
        minute.innerHTML = "00:";
        watchText.innerText = "reset";
    }

    function mode() {
        watchText.innerText = "mode is not effect";
    }
